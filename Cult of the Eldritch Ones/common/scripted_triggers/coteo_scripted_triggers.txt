    NOT = { is_in_society = cult_of_eldritch }

has_any_investigation_fishing_village_occurance = {
  OR = {
    has_character_flag = investigation_fishing_village_idol_taken
    has_character_flag = investigation_fishing_village_idol_broken
    has_character_flag = investigation_fishing_village_book_taken
    has_character_flag = investigation_fishing_village_book_burned
    has_character_flag = investigation_fishing_village_altar
  }
}

demesne_eldritch_holy_site = {
  OR = {
    AND = {
      lower_tier_than = king
      any_realm_title = {
        is_holy_site = eldritch_gods
      }
    }
    AND = {
      higher_tier_than = duke
      ## directly held demesne counties
      any_demesne_title = {
        is_holy_site = eldritch_gods
      }
    }
  }
}
